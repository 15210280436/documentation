ggplot(dat, aes(x=cond, y=rating)) + geom_boxplot() +

library(plotly)

y.1<-rnorm(100)

dfmelt<-melt(df, measure.vars = 2:5)

ggplot(dfmelt, aes(x=factor(round_any(x,0.5)), y=value,fill=variable))+
    geom_boxplot()+
    facet_grid(.~variable)+
    labs(x="X (binned)")+
    theme(axis.text.x=element_text(angle=-90, vjust=0.4,hjust=1))
