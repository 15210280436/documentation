--- name: using geom_line with geom_ribbon plot_url: https://plot.ly/~MattSundquist/17629/y-vs-x/ language: ggplot2 suite: geom_line description: Inspired by Stack Overflow. order: 9 sitemap: false --- # Learn about API authentication here: https://plot.ly/ggplot2/getting-started # Find your api_key here: https://plot.ly/settings/api library(plotly) set.seed(1) y <- sin(seq(1, 2*pi, length.out = 100)) x <- 1:100 plotdata <- data.frame(x=x, y=y, lower = (y+runif(100, -1, -0.5)), upper = (y+runif(100, 0.5, 1))) ggplot(plotdata) + geom_line(aes(y=y, x=x, colour = "sin"))+ geom_ribbon(aes(ymin=lower, ymax=upper, x=x, fill = "band"), alpha = 0.3)+ scale_colour_manual("",values="blue")+ scale_fill_manual("",values="grey12") ggplotly()
