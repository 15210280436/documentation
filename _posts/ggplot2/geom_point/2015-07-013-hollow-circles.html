--- name: Basic Scatter Chart Inspired By Winston Chang's [Cookbook](http://www.cookbook-r.com/Graphs/Scatterplots_(ggplot2)/) plot_url: https://plot.ly/~MattSundquist/16654 language: R suite: geom_point order: 1 --- # Learn about API authentication here: https://plot.ly/ggplot2/getting-started # Find your api_key here: https://plot.ly/settings/api set.seed(955) # Make some noisily increasing data dat <- data.frame(cond = rep(c("A", "B"), each=10), xvar = 1:20 + rnorm(20,sd=3), yvar = 1:20 + rnorm(20,sd=3)) head(dat) library(ggplot2) ggplot(dat, aes(x=xvar, y=yvar)) + geom_point(shape=1) # Use hollow circles py$ggplotly()
