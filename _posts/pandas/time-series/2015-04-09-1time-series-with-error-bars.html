--- name: Time Series with Error Bars plot_url: https://plot.ly/~jackp/10497/ language: pandas suite: time-series order: 1 sitemap: false arrangement: horizontal --- # Learn about API authentication here: https://plot.ly/pandas/getting-started # Find your api_key here: https://plot.ly/settings/api import plotly.plotly as py from plotly.graph_objs import * import pandas as pd df = pd.read_csv('https://raw.githubusercontent.com/plotly/datasets/master/wind_speed_laurel_nebraska.csv') df.head() upper_bound = Scatter( name='Upper Bound', x=df['Time'], y=df['10 Min Sampled Avg']+df['10 Min Std Dev'], mode='lines', marker=Marker(color="444"), line=Line(width=0), fillcolor='rgba(68, 68, 68, 0.3)', fill='tonexty' ) trace = Scatter( name='Measurement', x=df['Time'], y=df['10 Min Sampled Avg'], mode='lines', line=Line(color='rgb(31, 119, 180)'), fillcolor='rgba(68, 68, 68, 0.3)', fill='tonexty' ) lower_bound = Scatter( name='Lower Bound', x=df['Time'], y=df['10 Min Sampled Avg']-df['10 Min Std Dev'], marker=Marker(color="444"), line=Line(width=0), mode='lines' ) # Trace order can be important # with continuous error bars data = Data([lower_bound, trace, upper_bound]) layout = Layout( yaxis=YAxis(title='Wind speed (m/s)'), title='Continuous, variable value error bars
Notice the hover text!', showlegend = False ) fig = Figure( data=data, layout=layout ) # IPython notebook # py.iplot(fig, filename='pandas-time-series-error-bars') url = py.plot(fig, filename='pandas-time-series-error-bars')
