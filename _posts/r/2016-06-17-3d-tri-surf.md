# Basic Tri-Surf Plot

```r
library(plotly)

x = c(0, 1, 2, 0)
y = c(0, 0, 1, 2)
z = c(0, 2, 0, 1)
i = c(0, 0, 0, 1)
j = c(1, 2, 3, 2)
k = c(2, 3, 1, 3)
intensity = c(0, 0.33, 0.66, 1)
facecolor = list(
  'rgb(50, 255, 200)',
  'rgb(100, 255, 255)',
  'rgb(150, 255, 115)',
  'rgb(200, 255, 50)'
  )
)

p <- plot_ly(x = x, y = y, z = z,
             i = i, j = j, k = k,
             type = "mesh3d",
             facecolor = facecolor,
             showscale = TRUE
)
p
```
