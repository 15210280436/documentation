--- name: Parametric Plot plot_url: https://plot.ly/~tarzzz/215/parametric-plot/ language: python suite: 3d-parametric order: 0 sitemap: false --- import plotly.plotly as py from plotly.graph_objs import * import numpy as np s = np.linspace(0, 2 * np.pi, 240) t = np.linspace(0, np.pi, 240) tGrid, sGrid = np.meshgrid(s, t) r = 2 + np.sin(7 * sGrid + 5 * tGrid) # r = 2 + sin(7s+5t) x = r * np.cos(sGrid) * np.sin(tGrid) # x = r*cos(s)*sin(t) y = r * np.sin(sGrid) * np.sin(tGrid) # y = r*sin(s)*sin(t) z = r * np.cos(tGrid) # z = r*cos(t) surface = Surface(x=x, y=y, z=z) data = Data([surface]) layout = Layout( title='Parametric Plot', scene=Scene( xaxis=XAxis( gridcolor='rgb(255, 255, 255)', zerolinecolor='rgb(255, 255, 255)', showbackground=True, backgroundcolor='rgb(230, 230,230)' ), yaxis=YAxis( gridcolor='rgb(255, 255, 255)', zerolinecolor='rgb(255, 255, 255)', showbackground=True, backgroundcolor='rgb(230, 230,230)' ), zaxis=ZAxis( gridcolor='rgb(255, 255, 255)', zerolinecolor='rgb(255, 255, 255)', showbackground=True, backgroundcolor='rgb(230, 230,230)' ) ), autosize=False, width=800, height=800 ) fig = Figure(data=data, layout=layout) plot_url = py.plot(fig, filename='Parametric_plot')
