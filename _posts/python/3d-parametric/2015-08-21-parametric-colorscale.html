--- name: Parametric Plot With Colorscale plot_url: https://plot.ly/~tarzzz/318/another-parametric-plot/ language: python suite: 3d-parametric order: 1 sitemap: false --- import plotly.plotly as py from plotly.graph_objs import * import numpy as np dphi, dtheta = np.pi / 250.0, np.pi / 250.0 [phi, theta] = np.mgrid[0:np.pi + dphi * 1.5:dphi, 0:2 * np.pi + dtheta * 1.5:dtheta] m0 = 4; m1 = 3; m2 = 2; m3 = 3; m4 = 6; m5 = 2; m6 = 6; m7 = 4; # Applying the parametric equation.. r = np.sin(m0 * phi) ** m1 + np.cos(m2 * phi) ** m3 + np.sin(m4 * theta) ** m5 + np.cos(m6 * theta) ** m7 x = r * np.sin(phi) * np.cos(theta) y = r * np.cos(phi) z = r * np.sin(phi) * np.sin(theta) surface = Surface(x=x, y=y, z=z, colorscale='Viridis') data = Data([surface]) layout = Layout( title='Another Parametric Plot', scene=Scene( xaxis=XAxis( gridcolor='rgb(255, 255, 255)', zerolinecolor='rgb(255, 255, 255)', showbackground=True, backgroundcolor='rgb(230, 230,230)' ), yaxis=YAxis( gridcolor='rgb(255, 255, 255)', zerolinecolor='rgb(255, 255, 255)', showbackground=True, backgroundcolor='rgb(230, 230,230)' ), zaxis=ZAxis( gridcolor='rgb(255, 255, 255)', zerolinecolor='rgb(255, 255, 255)', showbackground=True, backgroundcolor='rgb(230, 230,230)' ) ), autosize=False, width=800, height=800 ) fig = Figure(data=data, layout=layout) plot_url = py.plot(fig, file_name='parametric-plot-viridis')
