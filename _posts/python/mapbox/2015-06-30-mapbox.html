import plotly.plotly as py
from plotly.graph_objs import *

data = Data([
    Scattermapbox(
        lat=['45.5017'],
        lon=['-73.5673'],
        mode='markers',
        text=['Montreal'],
    )
])
layout = Layout(
    autosize=True,
    hovermode='closest',
    mapbox=dict(
        bearing=0,
        center=dict(
            lat=45,
            lon=-73
        ),
        pitch=0,
        zoom=5
    ),
)

fig = Figure(data=data, layout=layout)
py.plot(fig, filename='Montreal Mapbox')
