def makeAxis(title): 
    return {
            'title': title,
            'ticksuffix': '%',
            'min': 0.01,
            'linewidth': 2,
            'ticks': 'outside',
            'ticklen': 8,
            'showgrid': true,
    }

    var data = Object.keys(rawData).map(function(k) {
        var pts = rawData[k];

        return {
            type: 'scatterternary',
            mode: 'lines',
            name: k,
            a: pts.map(function(d) { return d.clay }),
            b: pts.map(function(d) { return d.sand }),
            c: pts.map(function(d) { return d.silt }),
            line: { color: '#c00' }
        };
    });
