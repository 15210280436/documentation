Basic Line Plot

# Learn about API authentication here: https://plot.ly/julia/getting-started
# Find your api_key here: https://plot.ly/settings/api

using Plotly


trace1 = [
  "x" => [1, 2, 3, 4],
  "y" => [10, 15, 13, 17],
  "type" => "scatter"
]
trace2 = [
  "x" => [1, 2, 3, 4],
  "y" => [16, 5, 11, 9],
  "type" => "scatter"
]
data = [trace1, trace2]
response = Plotly.plot(data, ["filename" => "basic-line", "fileopt" => "overwrite"])
plot_url = response["url"]

Line and Scatter Plot

# Learn about API authentication here: https://plot.ly/julia/getting-started
# Find your api_key here: https://plot.ly/settings/api

using Plotly


trace1 = [
  "x" => [1, 2, 3, 4],
  "y" => [10, 15, 13, 17],
  "mode" => "markers",
  "type" => "scatter"
]
trace2 = [
  "x" => [2, 3, 4, 5],
  "y" => [16, 5, 11, 9],
  "mode" => "lines",
  "type" => "scatter"
]
trace3 = [
  "x" => [1, 2, 3, 4],
  "y" => [12, 9, 15, 12],
  "mode" => "lines+markers",
  "type" => "scatter"
]
data = [trace1, trace2, trace3]
response = Plotly.plot(data, ["filename" => "line-scatter", "fileopt" => "overwrite"])
plot_url = response["url"]
