var plotly = require('plotly')('username', 'api_key');

var x = [];

for (var i = 0; i &lt; 500; i ++) {
	x[i] = Math.random();
}


var data = [
  {
    x: x, 
    type: "histogram"
  }
];
var graphOptions = {filename: "basic-histogram", fileopt: "overwrite"};
plotly.plot(data, graphOptions, function (err, msg) {
    console.log(msg);
});
