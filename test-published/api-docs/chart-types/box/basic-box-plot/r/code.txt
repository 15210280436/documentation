library(plotly)
py &lt;- plotly(username='username', key='api_key')

y0 = rnorm(50)
y1 = rnorm(50)+1

trace1 &lt;- list(
  y = y0, 
  type = "box"
)
trace2 &lt;- list(
  y = y1, 
  type = "box"
)
data &lt;- list(trace1, trace2)
response &lt;- py$plotly(data, kwargs=list(filename="basic-box-plot", fileopt="overwrite"))
url &lt;- response$url
