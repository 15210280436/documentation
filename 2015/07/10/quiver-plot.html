# Learn about API authentication here: https://plot.ly/python/getting-started # Find your api_key here: https://plot.ly/settings/api import plotly.plotly as py import plotly.tools as tls from plotly.graph_objs import * import numpy as np x,y = np.meshgrid(np.arange(0, 2, .2), np.arange(0, 2, .2)) u = np.cos(x)*y v = np.sin(x)*y #Create quiver quiver = tls.TraceFactory.create_quiver(x, y, u, v) # Plot data = Data([quiver]) fig=Figure(data=data) url = py.plot( fig, filename='Quiver Plot Example' )
