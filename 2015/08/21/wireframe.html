import plotly.plotly as py from plotly.graph_objs import * import numpy as np # Creating the data x = np.linspace(-5, 5, 50) y = np.linspace(-5, 5, 50) xGrid, yGrid = np.meshgrid(y, x) R = np.sqrt(xGrid ** 2 + yGrid ** 2) z = np.sin(R) # Creating the plot lines = [] line_marker = Line(color='#0066FF', width=2) for i, j, k in zip(xGrid, yGrid, z): lines.append(Scatter3d(x=i, y=j, z=k, mode='lines', line=line_marker)) data = Data(lines) layout = Layout( title='Wireframe Plot', scene=Scene( xaxis=XAxis( gridcolor='rgb(255, 255, 255)', zerolinecolor='rgb(255, 255, 255)', showbackground=True, backgroundcolor='rgb(230, 230,230)' ), yaxis=YAxis( gridcolor='rgb(255, 255, 255)', zerolinecolor='rgb(255, 255, 255)', showbackground=True, backgroundcolor='rgb(230, 230,230)' ), zaxis=ZAxis( gridcolor='rgb(255, 255, 255)', zerolinecolor='rgb(255, 255, 255)', showbackground=True, backgroundcolor='rgb(230, 230,230)' ) ), showlegend=False, ) fig = Figure(data=data, layout=layout) plot_url = py.plot(fig, filename='wireframe_plot')
