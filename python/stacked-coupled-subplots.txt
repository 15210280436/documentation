import plotly
py = plotly.plotly('PythonAPI', 'ubpiol2cve')

data = [{'x': [0,1,2], 'y': [10,11,12]},
        {'x': [2,3,4], 'y': [100,110,120], 'yaxis': 'y2'},
        {'x': [3,4,5], 'y': [1000,1100,1200], 'yaxis': 'y3'}]
layout={
    'yaxis': {'domain': [0,1./3.]},
    'yaxis2': {'domain':[1./3,2./3.]},
    'yaxis3': {'domain':[2./3.,1]},
    'legend': {'traceorder': 'reversed'}
}

py.plot(data,layout=layout)
