# Learn about API authentication here: https://plot.ly/python/getting-started

import plotly.plotly as py
from plotly.graph_objs import *

trace1 = Bar(
    y=['giraffes', 'orangutans', 'monkeys'],
    x=[20, 14, 23],
    name='SF Zoo',
    orientation = 'h',
    marker = Marker(
        color = 'rgba(55, 128, 191, 0.6)',
        line = Line(
            color = 'rgba(55, 128, 191, 1.0)',
            width = 1,
        )
    )
)
trace2 = Bar(
    y=['giraffes', 'orangutans', 'monkeys'],
    x=[12, 18, 29],
    name='LA Zoo',
    orientation = 'h',
    marker = Marker(
        color = 'rgba(255, 153, 51, 0.6)',
        line = Line(
            color = 'rgba(255, 153, 51, 1.0)',
            width = 1,
        )
    )
)
data = Data([trace1, trace2])
layout = Layout(
    barmode='stack'
)
fig = Figure(data=data, layout=layout)
plot_url = py.plot(fig, filename='marker-h-bar')

Bar Chart with Line Plot

import plotly.plotly as py
import numpy as np
from plotly.graph_objs import *
from plotly import tools

y_saving = [1.3586, 2.2623000000000002, 4.9821999999999997, 6.5096999999999996,
            7.4812000000000003, 7.5133000000000001, 15.2148, 17.520499999999998
            ]
y_net_worth = [93453.919999999998, 81666.570000000007, 69889.619999999995,
               78381.529999999999, 141395.29999999999, 92969.020000000004,
               66090.179999999993, 122379.3]
x_saving = ['Japan', 'United Kingdom', 'Canada', 'Netherlands',
            'United States', 'Belgium', 'Sweden', 'Switzerland']
x_net_worth = ['Japan', 'United Kingdom', 'Canada', 'Netherlands',
               'United States', 'Belgium', 'Sweden', 'Switzerland'
               ]
trace0 = Bar(
    x=y_saving,
    y=x_saving,
    marker=Marker(
        color='rgba(50, 171, 96, 0.6)',
        line=Line(
            color='rgba(50, 171, 96, 1.0)',
            width=1,
        ),
    ),
    name='Household savings, percentage of household disposable income',
    orientation='h',
)
trace1 = Scatter(
    x=y_net_worth,
    y=x_net_worth,
    mode='lines+markers',
    line=Line(
        color='rgb(128, 0, 128)',
    ),
    name='Household net worth, Million USD/capita',
)
layout = dict(
    title='Household savings & net worth for eight OECD countries',
    yaxis1=YAxis(
        showgrid=False,
        showline=False,
        showticklabels=True,
        domain=[0, 0.85],
    ),
    yaxis2=YAxis(
        showgrid=False,
        showline=True,
        showticklabels=False,
        linecolor='rgba(102, 102, 102, 0.8)',
        linewidth=2,
        domain=[0, 0.85],
    ),
    xaxis1=XAxis(
        zeroline=False,
        showline=False,
        showticklabels=True,
        showgrid=True,
        domain=[0, 0.42],
    ),
    xaxis2=XAxis(
        zeroline=False,
        showline=False,
        showticklabels=True,
        showgrid=True,
        domain=[0.47, 1],
        side='top',
        dtick=25000,
    ),
    legend=Legend(
        x=0.029,
        y=1.038,
        font=Font(
            size=10,
        ),
    ),
    margin=Margin(
        l=100,
        r=20,
        t=70,
        b=70,
    ),
    width=600,
    height=600,
    paper_bgcolor='rgb(248, 248, 255)',
    plot_bgcolor='rgb(248, 248, 255)',
)

annotations = []

y_s = np.round(y_saving, decimals=2)
y_nw = np.rint(y_net_worth)

# Adding labels
for ydn, yd, xd in zip(y_nw, y_s, x_saving):
    # labeling the scatter savings
    annotations.append(Annotation(xref='x2', yref='y2', y=xd, x=ydn - 20000,
                                  text='{:,}'.format(ydn) + 'M',
                                  font=Font(family='Arial', size=12,
                                            color='rgb(128, 0, 128)'),
                                  showarrow=False,))
    # labeling the bar net worht
    annotations.append(Annotation(xref='x1', yref='y1', y=xd, x=yd + 3,
                                  text=str(yd) + '%',
                                  font=Font(family='Arial', size=12,
                                            color='rgb(50, 171, 96)'),
                                  showarrow=False,))
# Source
annotations.append(Annotation(xref='paper', yref='paper', x=-0.2, y=-0.109,
                              text='OECD "' +
                              '(2015), Household savings (indicator), ' +
                              'Household net worth (indicator). doi: ' +
                              '10.1787/cfc6f499-en (Accessed on 05 June 2015)',
                              font=Font(family='Arial', size=10,
                                        color='rgb(150,150,150)'),
                              showarrow=False,))

layout['annotations'] = annotations

# Creating two subplots
fig = tools.make_subplots(rows=1, cols=2, specs=[[{}, {}]], shared_xaxes=True,
                          shared_yaxes=False, vertical_spacing=0.001)

fig.append_trace(trace0, 1, 1)
fig.append_trace(trace1, 1, 2)

fig['layout'].update(layout)
plot_url = py.plot(fig, filename='oecd-networth-saving-bar-line')

Color Palette for Bar Chart

import plotly.plotly as py
from plotly.graph_objs import *

top_labels = ['Strongly<br>agree', 'Agree', 'Neutral', 'Disagree',
              'Strongly<br>disagree']

colors = ['rgba(38, 24, 74, 0.8)', 'rgba(71, 58, 131, 0.8)',
          'rgba(122, 120, 168, 0.8)', 'rgba(164, 163, 204, 0.85)',
          'rgba(190, 192, 213, 1)']

x_data = [[21, 30, 21, 16, 12],
          [24, 31, 19, 15, 11],
          [27, 26, 23, 11, 13],
          [29, 24, 15, 18, 14]]

y_data = ['The course was effectively<br>organized',
          'The course developed my<br>abilities and skills ' +
          'for<br>the subject', 'The course developed ' +
          'my<br>ability to think critically about<br>the subject',
          'I would recommend this<br>course to a friend']


traces = []

for i in range(0, len(x_data[0])):
    for xd, yd in zip(x_data, y_data):
        traces.append(Bar(
            x=xd[i],
            y=yd,
            orientation='h',
            marker=Marker(
                color=colors[i],
                line=Line(
                        color='rgb(248, 248, 249)',
                        width=1,
                )
            )
        ))

layout = Layout(
    xaxis=XAxis(
        showgrid=False,
        showline=False,
        showticklabels=False,
        zeroline=False,
        domain=[0.15, 1]
    ),
    yaxis=YAxis(
        showgrid=False,
        showline=False,
        showticklabels=False,
        zeroline=False,
    ),
    barmode='stack',
    height=600,
    width=600,
    paper_bgcolor='rgb(248, 248, 255)',
    plot_bgcolor='rgb(248, 248, 255)',
    margin=Margin(
        l=120,
        r=10,
        t=140,
        b=80
    ),
    showlegend=False,
)

annotations = []

for yd, xd in zip(y_data, x_data):
    # labeling the y-axis
    annotations.append(Annotation(xref='paper', yref='y', x=0.14, y=yd,
                                  xanchor='right', text=str(yd),
                                  font=Font(family='Arial', size=14,
                                            color='rgb(67, 67, 67)'),
                                  showarrow=False, align='right'))
    # labeling the first percentage of each bar (x_axis)
    annotations.append(Annotation(xref='x', yref='y', x=(xd[0] / 2), y=yd,
                                  text=str(xd[0]) + '%',
                                  font=Font(family='Arial', size=14,
                                            color='rgb(248, 248, 255)'),
                                  showarrow=False))
    # labeling the first Likert scale (on the top)
    if yd == y_data[-1]:
        annotations.append(Annotation(xref='x', yref='paper', x=(xd[0] / 2),
                                      y=1.1, text=top_labels[0],
                                      font=Font(family='Arial', size=14,
                                      color='rgb(67, 67, 67)'),
                                      showarrow=False))
    space = xd[0]
    for i in range(1, len(xd)):
            # labeling the rest of percentages for each bar (x_axis)
            annotations.append(Annotation(xref='x', yref='y', x=space
                                          + (xd[i] / 2), y=yd,
                                          text=str(xd[i]) + '%',
                                          font=Font(family='Arial', size=14,
                                                    color='rgb(248, 248, 255)'),
                                          showarrow=False))
            # labeling the Likert scale
            if yd == y_data[-1]:
                annotations.append(Annotation(xref='x', yref='paper', x=space +
                                              (xd[i] / 2), y=1.1,
                                              text=top_labels[i],
                                              font=Font(family='Arial', size=14,
                                                        color='rgb(67, 67, 67)'),
                                              showarrow=False))
            space += xd[i]

layout['annotations'] = annotations

fig = Figure(data=traces, layout=layout)

plot_url = py.plot(fig, filename='bar-colorscale')
