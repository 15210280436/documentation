library(plotly)
py &lt;- plotly(username='username', key='api_key')


trace1 &lt;- list(
  x = c(0, 1, 2), 
  y = c(1, 2, 3), 
  name = "First Trace", 
  showlegend = FALSE, 
  type = "scatter"
)
trace2 &lt;- list(
  x = c(0, 1, 2, 3), 
  y = c(8, 4, 2, 0), 
  name = "Second Trace", 
  showlegend = TRUE, 
  type = "scatter"
)
data &lt;- list(trace1, trace2)
response &lt;- py$plotly(data, kwargs=list(filename="show-legend", fileopt="overwrite"))
url &lt;- response$url
