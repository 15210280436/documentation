# Learn about API authentication here: plot.ly/r/getting-started
# Find your api_key here: plot.ly/settings/api

library(plotly)
py &lt;- plotly(username='username', key='api_key')

x0 = rnorm(500)
x1 = rnorm(500)+1

trace1 &lt;- list(
  x = x0, 
  type = "histogram"
)
trace2 &lt;- list(
  x = x1, 
  type = "histogram"
)
data &lt;- list(trace1, trace2)
layout &lt;- list(barmode = "stack")
response &lt;- py$plotly(data, kwargs=list(layout=layout, filename="stacked-histogram", fileopt="overwrite"))
url &lt;- response$url
